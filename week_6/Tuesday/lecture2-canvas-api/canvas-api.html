What is the Canvas API?

An HTML5 Drawing Surface

The Canvas API allows you to draw graphics using JavaScript:

<!-- HTML: Create a canvas element --> <canvas id="myCanvas" width="500" height="400"></canvas> // JavaScript: Get the canvas and context let canvas = document.getElementById("myCanvas"); let ctx = canvas.getContext("2d"); // Now you can draw on ctx!

Key Point: The canvas is a blank surface. You need JavaScript to draw on it using the 2D rendering context (ctx).

Drawing Rectangles

Basic Shape Methods

let canvas = document.getElementById("myCanvas"); let ctx = canvas.getContext("2d"); // Filled rectangle ctx.fillStyle = "orange"; ctx.fillRect(50, 50, 200, 100); // fillRect(x, y, width, height) // Outlined rectangle ctx.strokeStyle = "purple"; ctx.lineWidth = 5; ctx.strokeRect(300, 50, 150, 100); // Clear a rectangle area ctx.clearRect(100, 75, 50, 50);

Coordinate System: (0, 0) is at the top-left corner. X increases right, Y increases down.

Drawing Paths and Lines

Creating Custom Shapes

// Drawing a triangle ctx.beginPath(); // Start a new path ctx.moveTo(100, 50); // Move to starting point ctx.lineTo(150, 150); // Draw line to point ctx.lineTo(50, 150); // Draw line to point ctx.closePath(); // Connect back to start ctx.fillStyle = "#ff6b35"; ctx.fill(); // Fill the shape // Drawing a line ctx.beginPath(); ctx.moveTo(200, 50); ctx.lineTo(300, 150); ctx.strokeStyle = "black"; ctx.lineWidth = 3; ctx.stroke(); // Draw the outline

Drawing Circles and Arcs

Using the arc() Method

// Draw a full circle ctx.beginPath(); ctx.arc(250, 200, 50, 0, Math.PI * 2); // arc(x, y, radius, startAngle, endAngle) ctx.fillStyle = "#8b2b99"; ctx.fill(); // Draw a semi-circle (arc) ctx.beginPath(); ctx.arc(400, 200, 50, 0, Math.PI); // 0 to Math.PI = half circle (180 degrees) ctx.strokeStyle = "orange"; ctx.lineWidth = 3; ctx.stroke();

Angles: Measured in radians. 2π (Math.PI * 2) = 360°, π = 180°

Colors, Gradients, and Text

Styling Your Canvas

// Solid colors ctx.fillStyle = "red"; // Color name ctx.fillStyle = "#ff6b35"; // Hex color ctx.fillStyle = "rgb(255, 107, 53)"; // RGB // Linear gradient let gradient = ctx.createLinearGradient(0, 0, 200, 0); gradient.addColorStop(0, "orange"); gradient.addColorStop(1, "purple"); ctx.fillStyle = gradient; ctx.fillRect(50, 50, 200, 100); // Drawing text ctx.font = "30px Arial"; ctx.fillStyle = "black"; ctx.fillText("Hello Canvas!", 100, 250); // Outlined text ctx.strokeText("Outlined", 100, 300);

Drawing Images

Adding Images to Canvas

// Load and draw an image let img = new Image(); img.src = "pumpkin.png"; img.onload = function() { // Draw image at position ctx.drawImage(img, 50, 50); // Draw with custom size ctx.drawImage(img, 200, 50, 100, 100); // drawImage(image, x, y, width, height) // Crop and draw specific part ctx.drawImage(img, 0, 0, 50, 50, // Source x, y, width, height 350, 50, 100, 100 // Dest x, y, width, height ); };

Important: Always wait for the image to load using onload before drawing!

Creating Animations

Using requestAnimationFrame()

let x = 0; let y = 200; let speed = 2; function animate() { // Clear the canvas ctx.clearRect(0, 0, canvas.width, canvas.height); // Draw the object ctx.fillStyle = "orange"; ctx.fillRect(x, y, 50, 50); // Update position x += speed; // Bounce off right edge if (x + 50 > canvas.width || x < 0) { speed = -speed; } // Request next frame requestAnimationFrame(animate); } animate(); // Start the animation

Complete Example

Bouncing Ball Animation

let canvas = document.getElementById("myCanvas"); let ctx = canvas.getContext("2d"); let ball = { x: 250, y: 200, radius: 20, dx: 3, // Velocity in x direction dy: 2 // Velocity in y direction }; function drawBall() { ctx.beginPath(); ctx.arc(ball.x, ball.y, ball.radius, 0, Math.PI * 2); ctx.fillStyle = "#ff6b35"; ctx.fill(); ctx.closePath(); } function update() { ctx.clearRect(0, 0, canvas.width, canvas.height); drawBall(); // Check boundaries and bounce if (ball.x + ball.radius > canvas.width || ball.x - ball.radius < 0) { ball.dx = -ball.dx; } if (ball.y + ball.radius > canvas.height || ball.y - ball.radius < 0) { ball.dy = -ball.dy; } ball.x += ball.dx; ball.y += ball.dy; requestAnimationFrame(update); } update();
